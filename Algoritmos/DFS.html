void DFS(grafo G, nodo w) {
    T = crearArbol();  // OPCIONAL

    // Creamos la pila, y apilamos el primer elemento
    S = crearPila();

    while(!esVacia(S)) {
        // Extraemos el nodo u de la pila S
        u = desapilar(S);
        // Exploramos el nodo seleccionado
        if(!explorado[u]) {
            explorado[u] = true;
            //  Agregamos la arista al Ã¡rbol DFS (OPCIONAL)
            if(u != w) {
                agregarArista(T, u, padreAux[u]);
                padre[u] = padreAux[u];
            }

            for(v in adyacencia[u]) {
                apilar(S, v);
                padreAux[v] = u;
            }
        }
    }
    eliminarPila(S);
}
