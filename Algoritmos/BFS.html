void BFS(grafo G, nodo s) {
    T = crearArbol();  // OPCIONAL
    // Inicializamos los vértices
    for(u in vértices(G)) {
        visitado[u] = false;
        distancia[u] = -1;
        padre[u] = NULL;
        agregarNodo(T, u);  // OPCIONAL
    }
    // Exploramos el primer nodo
    visitado[s] = true;
    distancia[s] = 0;
    Q = crearCola();
    encolar(Q, s);
    while(!esVacia(Q)) {
        // Extraemos el nodo u de la cola Q
        u = desencolar(Q);
        // Exploramos todos los nodos adyacentes a u
        for(v in adyacencia[u]) {
            if(!visitado[v]) {
                visitado[v] = true;
                distancia[v] = distancia[u] + 1;
                padre[v] = u;
                encolar(Q, v);
                //  Agregamos la arista al árbol BFS (OPCIONAL)
                agregarArista(T, v, padre[v]);
            }
        }
    }
    eliminarCola(Q);
}
